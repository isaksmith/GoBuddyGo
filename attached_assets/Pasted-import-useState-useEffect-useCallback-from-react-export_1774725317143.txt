import { useState, useEffect, useCallback } from "react";

export interface Mission {
  id: string; title: string; icon: string;
  difficulty: "easy" | "medium" | "hard";
  completed: boolean; enabled: boolean;
}

export interface AppState {
  childName: string; sessionActive: boolean;
  sessionStartTime: number | null; sessionDuration: number;
  difficulty: "easy" | "medium" | "hard";
  missions: Mission[]; completedSessions: number;
  totalBadges: number; parentPin: string;
}

const DEFAULT_MISSIONS: Mission[] = [
  { id: "1", title: "Cheer for 5 seconds!", icon: "📣", difficulty: "easy", completed: false, enabled: true },
  { id: "2", title: "High-five the driver!", icon: "🖐️", difficulty: "easy", completed: false, enabled: true },
  { id: "3", title: "Count to 10 together!", icon: "🔢", difficulty: "medium", completed: false, enabled: true },
  { id: "4", title: "Do a silly dance!", icon: "💃", difficulty: "medium", completed: false, enabled: true },
  { id: "5", title: "Spot something blue!", icon: "🔵", difficulty: "hard", completed: false, enabled: true },
  { id: "6", title: "Make a funny face!", icon: "🤪", difficulty: "easy", completed: false, enabled: true },
  { id: "7", title: "Sing a song together!", icon: "🎵", difficulty: "hard", completed: false, enabled: true },
  { id: "8", title: "Give a thumbs up!", icon: "👍", difficulty: "easy", completed: false, enabled: true },
];

const STORAGE_KEY = "gobabygo-state";

const getInitialState = (): AppState => {
  try { const stored = localStorage.getItem(STORAGE_KEY); if (stored) return JSON.parse(stored); } catch {}
  return { childName: "", sessionActive: false, sessionStartTime: null, sessionDuration: 10, difficulty: "easy", missions: DEFAULT_MISSIONS, completedSessions: 0, totalBadges: 0, parentPin: "1234" };
};

export function useAppState() {
  const [state, setState] = useState<AppState>(getInitialState);
  useEffect(() => { localStorage.setItem(STORAGE_KEY, JSON.stringify(state)); }, [state]);

  const setChildName = useCallback((name: string) => setState(s => ({ ...s, childName: name })), []);
  const startSession = useCallback(() => setState(s => ({ ...s, sessionActive: true, sessionStartTime: Date.now(), missions: s.missions.map(m => ({ ...m, completed: false })) })), []);
  const endSession = useCallback(() => setState(s => ({ ...s, sessionActive: false, completedSessions: s.completedSessions + 1, totalBadges: s.totalBadges + s.missions.filter(m => m.completed && m.enabled).length })), []);
  const completeMission = useCallback((id: string) => setState(s => ({ ...s, missions: s.missions.map(m => m.id === id ? { ...m, completed: true } : m) })), []);
  const toggleMission = useCallback((id: string) => setState(s => ({ ...s, missions: s.missions.map(m => m.id === id ? { ...m, enabled: !m.enabled } : m) })), []);
  const setDifficulty = useCallback((d: "easy" | "medium" | "hard") => setState(s => ({ ...s, difficulty: d })), []);
  const setSessionDuration = useCallback((d: number) => setState(s => ({ ...s, sessionDuration: d })), []);

  return { state, setChildName, startSession, endSession, completeMission, toggleMission, setDifficulty, setSessionDuration };
}
