import { useState } from "react";
import splashBg from "@/assets/splash-bg.jpg";
import { useApp } from "@/context/AppContext";
import { useNavigate } from "react-router-dom";
import { Star, Shield, Settings } from "lucide-react";

const HomeScreen = () => {
  const { state, setChildName, startSession } = useApp();
  const [nameInput, setNameInput] = useState(state.childName);
  const navigate = useNavigate();

  const handleStart = () => {
    if (nameInput.trim()) { setChildName(nameInput.trim()); startSession(); navigate("/ar"); }
  };

  return (
    <div className="relative min-h-screen flex flex-col items-center justify-end overflow-hidden">
      <div className="absolute inset-0">
        <img src={splashBg} alt="GoBabyGo Splash" className="w-full h-full object-cover" width={768} height={1024} />
        <div className="absolute inset-0 bg-stripes" />
        <div className="absolute inset-0 bg-gradient-to-t from-background via-background/70 to-transparent" />
      </div>
      <div className="absolute top-20 left-8 animate-float"><div className="icon-circle w-10 h-10"><Star className="w-5 h-5 text-gold" fill="hsl(var(--gold))" /></div></div>
      <div className="absolute top-36 right-10 animate-float" style={{ animationDelay: "1s" }}><div className="icon-circle w-9 h-9"><Shield className="w-4 h-4 text-teal" fill="hsl(var(--teal))" /></div></div>
      <div className="absolute top-52 left-14 animate-float" style={{ animationDelay: "0.5s" }}><div className="icon-circle w-8 h-8"><Star className="w-4 h-4 text-coral" fill="hsl(var(--coral))" /></div></div>
      <div className="relative z-10 w-full px-6 pb-12 space-y-4">
        <div className="text-center mb-8">
          <h1 className="font-fredoka text-5xl text-foreground text-glow-coral mb-1">GO BUDDY GO</h1>
          <p className="font-fredoka text-lg text-teal text-glow-teal uppercase tracking-wider">Co-Pilot Adventure</p>
        </div>
        <div className="card-game p-3">
          <input type="text" placeholder="ENTER YOUR NAME, BUDDY!" value={nameInput} onChange={(e) => setNameInput(e.target.value)} className="w-full bg-input rounded-xl px-4 py-3.5 text-foreground font-fredoka text-center text-base placeholder:text-muted-foreground/60 placeholder:text-sm focus:outline-none focus:ring-2 focus:ring-coral uppercase" />
        </div>
        <button onClick={handleStart} disabled={!nameInput.trim()} className="btn-pill-primary w-full text-xl disabled:opacity-40 disabled:pointer-events-none">🚀 START MISSION</button>
        <button onClick={() => navigate("/parent")} className="btn-pill-outline w-full flex items-center justify-center gap-2"><Settings className="w-5 h-5" />PARENT MODE</button>
      </div>
    </div>
  );
};
export default HomeScreen;
